// Fitness Influencer Coaching Ecosystem - Industry Level ERD

users [icon: users, color: gray] {
	user_id string pk
	full_name string
	email string unique
	phone string
	password_hash string
	role string // ENUM: 'trainer', 'client', 'admin'
	status string // ENUM: 'active', 'inactive', 'blocked'
	created_at timestamp
	updated_at timestamp
}

	bio text
	specialization string

	certifications text
	pricing_currency string
	timezone string

// Relationships (Cardinality)
// 1 user -> 0..1 trainer profile
users.user_id < trainer_profiles.trainer_id

// 1 user -> 0..1 client profile
users.user_id < client_profiles.client_id

// 1 trainer -> many direct trainer-client links
trainer_profiles.trainer_id < coach_client_links.trainer_id

// 1 client -> many trainer links over time
client_profiles.client_id < coach_client_links.client_id

// 1 trainer -> many programs
trainer_profiles.trainer_id < programs.trainer_id

// 1 program -> many components
programs.program_id < program_components.program_id

// 1 client -> many subscriptions
client_profiles.client_id < subscriptions.client_id

// 1 program -> many subscriptions
programs.program_id < subscriptions.program_id

// 1 coach-client link -> many subscriptions over time
coach_client_links.link_id < subscriptions.coach_link_id

// 1 subscription -> many invoices
subscriptions.subscription_id < invoices.subscription_id

// 1 invoice -> many payments (supports partial/retry/refund flows)
invoices.invoice_id < payments.invoice_id

// 1 subscription -> many payments
subscriptions.subscription_id < payments.subscription_id

// 1 subscription -> many sessions
subscriptions.subscription_id < sessions.subscription_id

// 1 trainer -> many sessions
trainer_profiles.trainer_id < sessions.trainer_id

// 1 client -> many sessions
client_profiles.client_id < sessions.client_id

// 1 subscription -> many check-ins
subscriptions.subscription_id < check_ins.subscription_id

// 1 client -> many check-ins
client_profiles.client_id < check_ins.client_id

// 1 check-in -> many progress metric entries (extensible)
check_ins.check_in_id < progress_entries.check_in_id

// 1 check-in -> many trainer feedback records
check_ins.check_in_id < trainer_feedbacks.check_in_id

// 1 trainer -> many feedback records
trainer_profiles.trainer_id < trainer_feedbacks.trainer_id

// 1 subscription -> many goal milestones
subscriptions.subscription_id < goal_milestones.subscription_id

// 1 client -> many milestones
client_profiles.client_id < goal_milestones.client_id
